import numpy as np
import matplotlib.pyplot as plt

def line1(x):
    return -x + 4  # y = -x + 4

def line2(x):
    return 2*x - 2  # y = 2x - 2

x_values = np.linspace(-10, 10, 200)
y1 = line1(x_values)
y2 = line2(x_values)

plt.axhline(0, color='black', linewidth=0.8, linestyle="--")
plt.axvline(0, color='black', linewidth=0.8, linestyle="--")
plt.plot(x_values, y1, label="y = -x + 4", color="blue")
plt.plot(x_values, y2, label="y = 2x - 2", color="green")

plt.title("System 1: Unique solution at the point (2, 2)")
plt.xlabel("x")
plt.ylabel("y")
plt.xlim(-5, 5)
plt.ylim(-10, 10)
plt.grid(color='gray', linestyle='--', linewidth=0.5)
plt.legend()
plt.show()

def line1(x):
    return (-3/2)*x + 6  # y = -3/2 x + 6

def line2(x):
    return (-3/2)*x + 6  # same straight line

x_values = np.linspace(-5, 5, 200)
y1 = line1(x_values)
y2 = line2(x_values)

plt.axhline(0, color='black', linewidth=0.8, linestyle="--")
plt.axvline(0, color='black', linewidth=0.8, linestyle="--")
plt.plot(x_values, y1, label="y = -3/2x + 6", color="purple", linestyle="-")

# Draw the same line twice, it can also be done without line2, but to show that it is the same thing:
plt.plot(x_values, y2, label="the same line again", color="orange", linestyle="--")

plt.title("System 3: Infinite Solutions  Two Overlapping Lines")
plt.xlabel("x")
plt.ylabel("y")
plt.xlim(-5, 5)
plt.ylim(-5, 10)
plt.grid(color='gray', linestyle='--', linewidth=0.5)
plt.legend()
plt.show()
